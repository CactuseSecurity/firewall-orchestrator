
In general:
	- # is the comment sign (text will be ignored)
	- options separated by <space> except for gui.conf "usergroup <group> visible-clients:", where a "," is used


iso.conf
	# this file contains config options used across multiple modules (importer, web)
	# must be present on all installations
	
	ITSecOrgDir	/usr/share/itsecorg		# Installationsverzeichnis
	
	###################### database connection ##############################
	itsecorg database hostname              localhost
	itsecorg database type                  DBX_PGSQL
	itsecorg database name                  isodb
	itsecorg database port                  5432
	
	rule_header_offset	99000		# used for creating netscreen header_rules (containing zone-info)
									# (used in import and web-gui)
									# must at least be higher than max expected rule number
	
	###################### logging options ##################################
	set loglevel	7
	set logtarget	syslog
	syslog_local	6
	syslog_type		unix
	syslog_ident	ITSecOrg
	syslog_facility	local6

	# settings relevant for import modules only

	usergroup_delimiter	|

gui.conf
	# config parameters
	config password minimal-length 6

	###################### display options ##################################
	
	display max_number_of_rules_in_ruleview 3000
	display number_of_undocumented_changes 200
	display undocumented_rule_changes maxcols 30
	display undocumented_rule_changes maxrows 2
	
	###################### GUI Spracheinstellungen ##########################
	user tim language german
	user tom language english
	
	###################### User-Berechtigungen ##############################
	# group memberships, may have multiple lines
	usergroup isoadmins members: ralf tim hugo
	usergroup demo members: demo2 demo7 ssx dfs gost
	usergroup demo members: demo3 demo5
	
	# group privileges
	usergroup demo privileges:			view-reports document-changes view-change-admin-names change-documentation
	usergroup testminimum privileges:	view-reports
	usergroup isoadmins privileges:     view-reports document-changes admin-users admin-devices admin-clients change-documentation view-all-objects-filter
	
	# client visibility (NB: separator is here ',' not space!)
	# usergroup demo visible-clients:			'Client 1', 'Client 2'
	usergroup demo visible-clients:				ALL
	usergroup testminimum visible-clients:		ALL
	usergroup isoadmins visible-clients:		ALL
	
	# device and management systems visible in the GUI
	usergroup demo visible-managements:					demo-mgmt
	usergroup demo visible-devices:						ALL
	usergroup testminimum visible-managements:			mgmt1
	usergroup testminimum visible-devices:				dev1 dev2
	usergroup isoadmins visible-managements:			ALL
	usergroup isoadmins visible-devices:				ALL
	
	# report visibility
	# list of reports: 
	#	configs		(Konfiguration)				- standard config report 
	#	changes		(Änderungen)				- old change report lists changes chronologically
	#	usage		(Verwendung)				- lists #rules, # objects for a device
	#	search		(Regelsuche) 				- similar to configs but only showing "allowing rules", filtering out section headings and gives "search through all systems" functionality 
	#	audit		(Audit-Änderungen)			- Audit change report, showing changes within a single rulebase (using colors + text); shows only what has changed
	#	details		(Audit-Änderungen Details)	- same as audit but additionally allowing to apply filtering and to also view unchangd rules
	#	ALL										- shows all reports
	
	usergroup demo visible-reports:				configs usage search
	usergroup testminimum visible-reports:		changes audit details
	usergroup isoadmins visible-reports:		ALL
	
	# parameters for document changes
	docu-changes default-client demo: Revi 
	docu-changes default-client demo-minimum: Revi 
	docu-changes default-client isoadmins: Gold
	docu-changes default-request-type isoadmins: ARS
	docu-changes number-of-requests 1
	docu-changes display-approver 0
	docu-changes request-type 0
	docu-changes comment-is-mandatory 1
